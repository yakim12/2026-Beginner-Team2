#include <iostream>
#include <stack> // 스택 사용을 위해 반드시 필요

using namespace std;

int main() {
    int N;
    if (!(cin >> N)) return 0; // 명령의 개수 입력

    stack<int> s; // 정수형 스택 선언

    for (int i = 0; i < N; i++) {
        char cmd;
        cin >> cmd;

        if (cmd == 'i') {
            int num;
            cin >> num;
            s.push(num); // 데이터 삽입
        } 
        else if (cmd == 'o') {
            if (s.empty()) {
                cout << "empty" << endl; // 비어있으면 empty 출력
            } else {
                // top으로 확인하고 pop으로 제거 (2단계 과정)
                cout << s.top() << endl; 
                s.pop();
            }
        } 
        else if (cmd == 'c') {
            cout << s.size() << endl; // 저장된 데이터 개수
        }
    }

    return 0;
}
